% ── Hierarchical sample limitations slide ─────────────────────────────────────────────────────────────
\begin{frame}{Limitations of the Hierarchical Sampling Framework}
  \begin{itemize}
    \setlength\itemsep{1.0em}
    \item Each row of $\mathbf{X}_{n,m}$ has the law
      \[
        \mathbb{P}\!\big(X_{i,1}\in A_1,\ldots,X_{i,m}\in A_m\big)
        = \int_{\probx} \prod_{j=1}^{m} P(A_j)\, Q(dP),
        \qquad i=1,\ldots,n.
      \]
    \pause
    \item $\implies (X_{i,1},\ldots,X_{i,m})$ is exchangeable.
    \pause
    \item \textbf{Limitation.} $Q^1 \neq Q^2$ does \emph{not} imply
      $\mathbf{X}_{n,m} \overset{d}{\neq} \mathbf{Y}_{n,m}$.
    \pause
    \item \textbf{But for $m$ large:}
      \[
        Q^1 = Q^2
        \;\iff\;
        \mathbf{X}_{n,m} \overset{d}{=} \mathbf{Y}_{n,m}.
      \]
  \end{itemize}
\end{frame}

% ── Summary 1 slide ─────────────────────────────────────────────────────────────

\begin{frame}{Distance-based Two Sample Test}
  \begin{itemize}
    \setlength\itemsep{1.2em}
    \item \textbf{Problem.} $Q^1, Q^2 \in \probprobx$;\quad
          test $H_0 : Q^1 = Q^2$ vs $H_1 : Q^1 \neq Q^2$.
    \pause
    \item \textbf{Strategy.} Given hierarchical samples $\mathbf{X}_{n,m} \sim Q^1$
          and $\mathbf{Y}_{n,m} \sim Q^2$, reject $H_0$ if
          $d(\mathbf{X}_{n,m},\, \mathbf{Y}_{n,m})$ is large.
    \pause
    \item \textbf{Ingredients.} A distance $d$; a threshold.
  \end{itemize}
\end{frame}


% ── Hierarchical Empirical Estimator slide ─────────────────────────────────────────────────────────────

\begin{frame}{Hierarchical Empirical Estimator}
  \begin{itemize}
    \setlength\itemsep{1.1em}
    \item $Q \in \probprobx$. \pause
    \item Analogue of the empirical measure in i.i.d case:$\quad \widetilde{P}_1,\dots, \widetilde{P}_n \simiid Q$
      \[
        \frac{1}{n}\sum_{i=1}^{n} \delta_{\widetilde{P}_i}.
      \]
    
    \item But $\widetilde{P}_1,\ldots,\widetilde{P}_n$ are not observed.
    \pause
    \item Estimate each $\widetilde{P}_i$ from its samples:
    $\widehat{P}_{i,m} = \frac{1}{m}\sum_{j=1}^{m}\delta_{X_{i,j}}$ 
    \pause
    \item \textbf{Hierarchical Empirical Estimator:} 
    \[
        \widehat{Q}_{n,m} = \frac{1}{n}\sum_{i=1}^{n}\delta_{\widehat{P}_{i,m}}.
    \]
  \end{itemize}
\end{frame}

% ── Summary 2 slide ─────────────────────────────────────────────────────────────
\begin{frame}{Distance based Two Sample Test}
  \begin{itemize}
    \setlength\itemsep{1.1em}
    \item $Q^1, Q^2 \in \probprobx$;\quad test $H_0: Q^1 = Q^2$ vs $H_1: Q^1 \neq Q^2$.
    \item Observe hierarchical samples
      \[
        \mathbf{X}_{n,m} \sim Q^1, \qquad \mathbf{Y}_{n,m} \sim Q^2.
      \]
    \item Build hierarchical empirical estimators:
      \[
        \widehat{Q}^1_{n,m} = \frac{1}{n}\sum_{i=1}^{n}\delta_{\widehat{P}^1_{i,m}}, \qquad
        \widehat{Q}^2_{n,m} = \frac{1}{n}\sum_{i=1}^{n}\delta_{\widehat{P}^2_{i,m}}.
      \]
    \pause
    \item \textbf{Reject $H_0$} if
      \[
        d\!\left(\widehat{Q}^1_{n,m},\, \widehat{Q}^2_{n,m}\right) \text{ is large}.
      \]
  \end{itemize}
\end{frame}



% ── Wasserstein over Wasserstein slide ─────────────────────────────────────────────────────────────
\begin{frame}{Metrics on $\probprobx$: Wasserstein over Wasserstein}
  \begin{itemize}
    \setlength\itemsep{1em}
    \item $(\X, d_{\X})$ Polish space with bounded diameter.
    \item \textbf{Step 1:} Using $d_{\X}$ on $\X$, define Wasserstein distance $\mathcal{W}$ on $\probx$: for $P^1, P^2 \in \probx$,
      \[
        \mathcal{W}(P^1,P^2) = \inf_{(X^1,X^2)\in \Gamma(P^1,P^2)} \mathbb{E}\!\left[d_{\X}(X^1,X^2)\right],
      \]
      where $\Gamma(P^1,P^2)$ are couplings of $P^1$ and $P^2$ on $\X\times\X$.
    \pause
    \item \textbf{Step 2:} Using $\mathcal{W}$ on $\probx$, for $Q^1, Q^2 \in \probprobx$,
      \[
        \WW(Q^1,Q^2) = \inf_{(\widetilde{P}^1,\widetilde{P}^2)\sim \Gamma(Q^1,Q^2)} \mathbb{E}\!\left[\mathcal{W}(\widetilde{P}^1,\widetilde{P}^2)\right].
      \]
  \end{itemize}
  \vskip 0.8em
  \[
    \WW\ \text{is called} \textcolor{brown}{\textbf{ Wasserstein over Wasserstein (WoW).}}
  \]
\end{frame}

  \begin{itemize}
    \setlength\itemsep{0.4em}
    \item $\WW$ and $\dlip$ are \textbf{Integral Probability Metrics}: $\displaystyle \mathcal{I}_{\mathcal{G}}(\mu, \nu) = \sup_{g\in\mathcal{G}}\bigl|\mu(g)-\nu(g)\bigr|$.
    \pause
    \begin{align*}
      \WW(Q^1,Q^2) &= \sup_{h\in\mathrm{Lip}_1(\probx,\R,\W)}\!\left|\int_{\probx}h\,dQ^1 - \int_{\probx}h\,dQ^2\right|, \\[0.15cm]
      \dlip(Q^1,Q^2) &= \sup_{h\in\D}\left|\int_{\probx}h\,dQ^1 - \int_{\probx}h\,dQ^2\right|,
    \end{align*}
    where $\D = \bigl\{h:\probx\to\R \;:\; h(P) = g(P(f)),\; g\in\lipR,\; f\in\lipX\bigr\}$.
    \pause
    \item $\WW$ metrizes weak convergence on $\probprobx$.
    \item $\dlip$ metrizes weak convergence on $\probprobx$ when $\X$ is compact.
    \pause
    \item $\E\bigl[\WW(\widehat{Q}_{n,m}, Q)\bigr] \to 0$ \textbf{slower than} any polynomial rate.
    \item $\E\bigl[\dlip(\widehat{Q}_{n,m}, Q)\bigr] \to 0$ at \textbf{polynomial rate}.
  \end{itemize}
  \vskip 0.3em
  \hfill\parencite{catalano2024hierarchical}
\end{frame}



% ── Summary 3 slide ─────────────────────────────────────────────────────────────
\begin{frame}{Distance based Two Sample Test}
  \begin{itemize}
    \setlength\itemsep{1.1em}
    \item $Q^1, Q^2 \in \probprobx$;\quad test $H_0: Q^1 = Q^2$ vs $H_1: Q^1 \neq Q^2$.
    \item Observe hierarchical samples
      \[
        \mathbf{X}_{n,m} \sim Q^1, \qquad \mathbf{Y}_{n,m} \sim Q^2.
      \]
    \item Build hierarchical empirical estimators:
      \[
        \widehat{Q}^1_{n,m} = \frac{1}{n}\sum_{i=1}^{n}\delta_{\widehat{P}^1_{i,m}}, \qquad
        \widehat{Q}^2_{n,m} = \frac{1}{n}\sum_{i=1}^{n}\delta_{\widehat{P}^2_{i,m}}.
      \]
    \item \textbf{Reject $H_0$} if
      \[
        d\!\left(\widehat{Q}^1_{n,m},\, \widehat{Q}^2_{n,m}\right) \textcolor{red}{\text{ is large}}.
      \]
  \end{itemize}
\end{frame}






% ── Rademacher complexity threshold slide ─────────────────────────────────────
\begin{frame}{Theoretical Threshold via Rademacher Complexity}
  \begin{itemize}
    \item \textbf{Idea:} Threshold $r(n,m,\theta)$ from a concentration inequality.
  \end{itemize}
  \pause
  \vskip 0.6em
  \begin{theorem}
    Let $\mathcal{G}$, $\mathcal{H}$ be uniformly bounded function classes generating IPMs on $\probx$ and $\probprobx$ respectively, with $\mathcal{H} \subseteq \mathrm{Lip}_1(\probx,\R,\mathcal{I}_{\mathcal{G}})$. Then for all $\theta\in(0,1)$, $n,m\geq 1$,
    \[
      \mathbb{P}\!\left(\bigl|\mathcal{I}_{\mathcal{H}}(\widehat{Q}^1_{n,m},\widehat{Q}^2_{n,m}) - \mathcal{I}_{\mathcal{H}}(Q^1,Q^2)\bigr| \geq r(n,m,\theta)\right) \leq \theta,
    \]
    where
    \[
      r(n,m,\theta) = O\!\left(\mathcal{R}_n(\mathcal{H}) + \mathcal{R}_m(\mathcal{G}) + \frac{1}{\sqrt{n}}\right).
    \]
  \end{theorem}
\end{frame}

% ── Permutation threshold theorem slide ──────────────────────────────────
\begin{frame}{Practical Threshold via Permutation}
  \begin{itemize}
    \setlength\itemsep{0.9em}
    \onslide<1->{\item \textbf{Test statistic:}
      $T_{n,m} = \sqrt{\dfrac{n}{2}}\,\dlip\!\left(\widehat{Q}^1_{n,m},\, \widehat{Q}^2_{n,m}\right)$}
    \onslide<1->{\item \textbf{Permutation threshold:} for significance level $\theta\in(0,1)$,
      \[
        r(n,m,\theta) = \inf\!\left\{t\in\R : \mathbb{P}\!\left(\widetilde{T}_{n,m} > t \;\middle|\; \mathbf{X}_{n,m},\mathbf{Y}_{n,m}\right) \leq \theta\right\}.
      \]}
    \onslide<2->{\item \textbf{Testing scheme:} Reject $H_0$ if $T_{n,m} > r(n,m,\theta)$.}
  \end{itemize}
  \onslide<3->{
  \begin{theorem}
    Let $\X = [a,b]$ and $Q^1, Q^2 \in \probprobx$. Then for all $\theta \in (0,1)$
    \begin{itemize}
      \setlength\itemsep{0.4em}
      \item if $Q^1 = Q^2$:\; $\displaystyle\lim_{n,m\to\infty}\mathbb{P}\!\left(T_{n,m} > r(n,m,\theta)\right) \leq \theta$.
      \item if $Q^1 \neq Q^2$:\; $\displaystyle\lim_{n,m\to\infty}\mathbb{P}\!\left(T_{n,m} > r(n,m,\theta)\right) = 1$.
    \end{itemize}
  \end{theorem}
  }
\end{frame}
