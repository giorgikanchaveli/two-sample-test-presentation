\begin{frame}{Two-Sample Testing in $\R$}
  \begin{columns}[T]
    \begin{column}{0.48\textwidth}
      \onslide<4->{
      \textit{(unknown)} $P, Q \in \mathcal{P}(\R)$
      \[
        \textit{(Observed)}\quad X_1, \ldots, X_n \simiid P, \quad Y_1, \ldots, Y_n \simiid Q.
      \]
      Test $H_0 : P = Q$ vs $H_1 : P \neq Q$.
      }

\begin{frame}{Distance-based Two Sample Test}
  \begin{itemize}
    \setlength\itemsep{1.2em}
    \item \textbf{Problem.} $Q^1, Q^2 \in \probprobx$;\quad
          test $H_0 : Q^1 = Q^2$ vs $H_1 : Q^1 \neq Q^2$.
    \pause
    \item \textbf{Strategy.} Given hierarchical samples $\mathbf{X}_{n,m} \sim Q^1$
          and $\mathbf{Y}_{n,m} \sim Q^2$, reject $H_0$ if
          $d(\mathbf{X}_{n,m},\, \mathbf{Y}_{n,m})$ is large.
    \pause
    \item \textbf{Ingredients.} A distance $d$ on $\probprobx$; a threshold.
  \end{itemize}
\end{frame}

% ── Hierarchical sample limitations slide ─────────────────────────────────────────────────────────────
\begin{frame}{Limitations of the Hierarchical Sampling Framework}
  \begin{itemize}
    \setlength\itemsep{1.0em}
    \item Each row of $\mathbf{X}_{n,m}$ has the law
      \[
        \mathbb{P}\!\big(X_{i,1}\in A_1,\ldots,X_{i,m}\in A_m\big)
        = \int_{\probx} \prod_{j=1}^{m} P(A_j)\, Q(dP),
        \qquad i=1,\ldots,n.
      \]
    \pause
    \item $\implies (X_{i,1},\ldots,X_{i,m})$ is exchangeable.
    \pause
    \item \textbf{Limitation.} $Q^1 \neq Q^2$ does \emph{not} imply
      $\mathbf{X}_{n,m} \overset{d}{\neq} \mathbf{Y}_{n,m}$.
    \pause
    \item \textbf{But for $m$ large:}
      \[
        Q^1 = Q^2
        \;\iff\;
        \mathbf{X}_{n,m} \overset{d}{=} \mathbf{Y}_{n,m}.
      \]
  \end{itemize}
\end{frame}

% ── Hierarchical Empirical Estimator slide ─────────────────────────────────────────────────────────────

\begin{frame}{Hierarchical Empirical Estimator}
  \begin{itemize}
    \setlength\itemsep{1.1em}
    \item $Q \in \probprobx$. \pause
    \item Analogue of the empirical measure in i.i.d case:
      \[
        \frac{1}{n}\sum_{i=1}^{n} \delta_{\widetilde{P}_i}.
      \]
    
    \item But $\widetilde{P}_1,\ldots,\widetilde{P}_n$ are not observed.
    \pause
    \item Estimate each $\widetilde{P}_i$ from its samples:
    $\widehat{P}_{i,m} = \frac{1}{m}\sum_{j=1}^{m}\delta_{X_{i,j}}$ 
    \pause
    \item \textbf{Hierarchical Empirical Estimator:} 
    \[
        \widehat{Q}_{n,m} = \frac{1}{n}\sum_{i=1}^{n}\delta_{\widehat{P}_{i,m}}.
    \]
  \end{itemize}
\end{frame}

\begin{frame}{Distance based Two Sample Test}
  \begin{itemize}
    \setlength\itemsep{1.1em}
    \item $Q^1, Q^2 \in \probprobx$ unknown;\quad test $H_0: Q^1 = Q^2$ vs $H_1: Q^1 \neq Q^2$.
    \item Observe hierarchical samples
      \[
        \mathbf{X}_{n,m} \sim Q^1, \qquad \mathbf{Y}_{n,m} \sim Q^2.
      \]
    \item Build hierarchical empirical estimators:
      \[
        \widehat{Q}^1_{n,m} = \frac{1}{n}\sum_{i=1}^{n}\delta_{\widehat{P}^1_{i,m}}, \qquad
        \widehat{Q}^2_{n,m} = \frac{1}{n}\sum_{i=1}^{n}\delta_{\widehat{P}^2_{i,m}}.
      \]
    \pause
    \item \textbf{Reject $H_0$} if
      \[
        d\!\left(\widehat{Q}^1_{n,m},\, \widehat{Q}^2_{n,m}\right) \text{is large}.
      \]
  \end{itemize}
\end{frame}
